javascript定时器输出动态时间

发表于 2012-05-31   |   分类于 学习笔记

首先我们来介绍几个关于定时器的对象和属性：

1.setInterval（）

var a = window.setInterval(‘alert(1)’, 2000);
window.clearInterval(a);

setInterval（a，b）中的a代表要执行的内容，b中代表隔多久执行一次，注意：改对象的执行是循环的，类似于一个for循环，而且是一直循环的。

2.setTimeout（）

window.setTimeout(‘open(“http://www.baidu.com“)’, 2000);

setTimeout（a,b）中的a,b的意义和setInterval（）一样，但是和setInterval（）不同的是，setTimeout（）不会循环执行，意思就是这里的百度只会弹出一次。

下面我们来一个小例子：

// 在页面弹出1 2 3 。。 10，隔2秒钟弹出一个值，弹完10之后不就弹了

<script type=”text/javascript”>

var al,i=1;
al = window.setInterval(function(){
alert(i);
if(i>=10){
window.clearInterval(al);
}
i++;
},2000);

</script>

3.document.getElementById()

document.getElementById()是获取标签的id号，例如：

<div id=”timer2”></div>

document.getElementById(‘time2’);//获取time2的这个标签对应的id号

4.etElementsByTagName()

document.getElementsByTagName() 方法会使用指定的标签名返回所有的元素（作为一个节点列表），这些元素是您在使用此方法时所处的元素的后代（意思就是获取对应的标签名）例如：

document.getElementsByTagName(‘a’)获取所有的a标签

5. getElementsByName() 获取html中标签的name值 例如：

<p name=p1></p>

doxument.getElementsByName(‘p1’); 获取p标签中的name属性 注意：div标签是木有name属性的，甚用！

6.innerHTML 取得元素中的所有html代码

例如:

<h1 align=”center” id=”title”>这是标题1<span>333</span></h1>

alert(title.innerHTML);//弹出id为title的标签的HTML

// 定时器：输出动态时间 2012-05-28 21:17:30

<p id=”timer”></p>

<div id=”timer2”></div>

<script type=”text/javascript”>

function showTime(id){
document.getElementById(id).innerHTML = localeDate();
setTimeout(function(){showTime(id)},1000);
}

//获取本地时间 输出2012-05-28 21:17:30 这种的时间格式

function localeDate(){
var d = new Date();
var year = d.getFullYear();
var month = d.getMonth();
var date = d.getDate();
var hour = d.getHours();
var minute = d.getMinutes();
var second = d.getSeconds();
return year+’-‘+add0(month+1)+’-‘+add0(date)+’ ‘
+add0(hour)+’:’+add0(minute)+”:”+add0(second);
}
</script>

<script type=text/javascript>

// 冒泡法排序
var arr = [1,4,33,5,88,212,6];

function asort(arr, type){
var l = arr.length;
for(var i=0; i<l; i++){
for(var j=i+1; j<l; j++){

if(type == 1){
if(arr[i] > arr[j]){
var t = arr[i];
arr[i] = arr[j];
arr[j] = t;
}
}else{
if(arr[i] < arr[j]){
var t = arr[i];
arr[i] = arr[j];
arr[j] = t;
}
}
}
}
return arr;
}
alert(asort(arr, 2));

</script>

杨辉三角
1
1 1
1 2 1
1 3 3 1
1 4 6 4 1
1 5 10 10 5 1

分析:
1.每一行的第一个和最后一个为1
2.除了第一个和最后一个，当前的那个数等于它正上方和左上方的数相加
3.类似99乘法表
*/
// 新建一个数组
var n = 20;
var arr = [];
for(var k=0; k<n; k++){
arr[k] = [];
}

for(var j=0; j<=i; j++){

arr[i][j] = 1;
}else{
arr[i][j] = arr[i-1][j] + arr[i-1][j-1]
}
document.write(arr[i][j] + ‘&nbsp;’);
}
document.write(‘<br/>’);
}
